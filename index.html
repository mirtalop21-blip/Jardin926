(ausentes / activos * 100).toFixed(0) : 0; const esCritico = pctAusen > 25; // Alerta si falta más del 25% return `

${s.replace('SALA ','')}

${activos}/${LIMIT}
VAC: ${vacantes}
INASIST: ${pctAusen}%
`; }).join(''); } function renderAsis() { const grid = document.getElementById('gridAsis'); const activos = masterDB.filter(a => a.active); grid.innerHTML = activos.map(a => `

${a.n.split(',')[0]}

${a.sala}

`).join(''); const total = activos.length; const presentes = activos.filter(a => a.present).length; if(total > 0) { document.getElementById('txtPres').innerText = ((presentes/total)*100).toFixed(0) + "%"; document.getElementById('txtAus').innerText = (100 - (presentes/total)*100).toFixed(0) + "%"; } } function toggleAsis(id) { const idx = masterDB.findIndex(a => a.id === id); masterDB[idx].present = !masterDB[idx].present; sync(); } function toggleStatus(id) { const idx = masterDB.findIndex(a => a.id === id); masterDB[idx].active = !masterDB[idx].active; masterDB[idx].monthOut = masterDB[idx].active ? null : new Date().getMonth(); sync(); } function renderMovimientos() { const meses = ["ENERO","FEBRERO","MARZO","ABRIL","MAYO","JUNIO","JULIO","AGOSTO","SEPTIEMBRE","OCTUBRE","NOVIEMBRE","DICIEMBRE"]; const cont = document.getElementById('logBajas'); let html = ""; meses.forEach((m, i) => { const inCount = masterDB.filter(a => a.monthIn === i).length; const outCount = masterDB.filter(a => a.monthOut === i).length; if(inCount > 0 || outCount > 0) { html += `
${m}
+${inCount} ALTAS -${outCount} BAJAS
`; } }); cont.innerHTML = html || "

SIN MOVIMIENTOS HISTÓRICOS

"; } function renderTabla() { const body = document.getElementById('tableMain'); body.innerHTML = masterDB.map(a => ` ${a.n}
DNI: ${a.d} ${a.sala} ${a.active ? 'ACTIVO' : 'BAJA'} `).join(''); } function exportJSON() { const blob = new Blob([JSON.stringify(masterDB)], {type:'application/json'}); const a = document.createElement('a'); a.href = URL.createObjectURL(blob); a.download=`J926_MASTER_DB_${Date.now()}.json`; a.click(); } function importJSON(e) { const reader = new FileReader(); reader.onload = (ev) => { masterDB = JSON.parse(ev.target.result); sync(); alert("SINCRO COMPLETA"); }; reader.readAsText(e.target.files[0]); } function exportExcel() { let csv = "\ufeffALUMNO;DNI;SALA;ESTADO;INASISTENCIA\n"; masterDB.forEach(a => { csv += `${a.n};${a.d};${a.sala};${a.active?'ALTA':'BAJA'};${a.present?'NO':'SÍ'}\n`; }); const a = document.createElement('a'); a.href = 'data:text/csv;charset=utf-8,' + encodeURIComponent(csv); a.download="INFORME_MATRICULA_J926.csv"; a.click(); }
